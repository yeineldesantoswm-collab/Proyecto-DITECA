SEDE PRINCIPAL – R-CORE (SANTO DOMINGO)
! ===== FASE 1 – ISAKMP =====
crypto isakmp policy 10
encr aes 256
hash sha256
authentication pre-share
group 2
lifetime 86400

!Claves compartidas con las sucursales
crypto isakmp key VPNKEY address IP_PUBLICA_SANTIAGO
crypto isakmp key VPNKEY address IP_PUBLICA_ROMANA

!FASE 2 – IPsec
crypto ipsec transform-set TS-VPN esp-aes 256 esp-sha-hmac
mode tunnel

!Túnel SD ↔ SANTIAGO
crypto map CM-VPN 10 ipsec-isakmp
set peer IP_PUBLICA_SANTIAGO
set transform-set TS-VPN
match address VPN-SD-SG

! Túnel SD ↔ ROMANA
crypto map CM-VPN 20 ipsec-isakmp
set peer IP_PUBLICA_ROMANA
set transform-set TS-VPN
match address VPN-SD-RM

!Aplicar crypto map a la WAN
interface e0/0
crypto map CM-VPN
